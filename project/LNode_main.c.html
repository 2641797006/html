int choose(int n)
{
if(0x30<=n&&n<=0x39)
n-=0x30;
else if(0x41<=n&&n<=0x5a)
n-=0x40;
else if(0x61<=n&&n<=0x7a)
n-=0x60;
else
n=-1;
return n;
}
int ychoose(int n)

}
